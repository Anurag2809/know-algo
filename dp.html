Dynamic Programming

Break a problem into overlapping subproblems and cache the results. DP is the hardest interview topic — but once you recognize the patterns, most problems reduce to a handful of templates. This guide teaches you those patterns.

Fundamentals

What is Dynamic Programming?

DP applies when a problem has two properties:

  1. Optimal substructure: the optimal solution to the problem can be built from optimal solutions to subproblems
  2. Overlapping subproblems: the same subproblems are solved multiple times in a naive recursive solution
Fibonacci without DP: fib(5) ├── fib(4) │ ├── fib(3) │ │ ├── fib(2) ←── computed 3 times! │ │ └── fib(1) │ └── fib(2) ←── computed again └── fib(3) ←── computed again ├── fib(2) ←── and again └── fib(1) Time: O(2^n) without memoization With memoization: each fib(i) computed ONCE → O(n)
DP vs Greedy vs Divide & Conquer
  • Greedy: make the locally optimal choice at each step (no backtracking). Works when local optimum = global optimum.
  • Divide & Conquer: split into independent subproblems (Merge Sort). No overlapping subproblems.
  • DP: subproblems overlap and we cache results. When greedy fails, try DP.
Two Approaches

Memoization vs Tabulation

Python
# ── MEMOIZATION (Top-Down) ──────────────────────────────────────────
# Start with recursion, cache results.
# Natural to write, easier to reason about state.
# Space: O(n) for memo + O(n) call stack.

from functools import lru_cache

@lru_cache(maxsize=None)
def fib_memo(n):
    if n <= 1:
        return n
    return fib_memo(n-1) + fib_memo(n-2)

# Manual memo dict (same idea)
def fib_memo_manual(n, memo={}):
    if n in memo: return memo[n]
    if n <= 1: return n
    memo[n] = fib_memo_manual(n-1, memo) + fib_memo_manual(n-2, memo)
    return memo[n]


# ── TABULATION (Bottom-Up) ──────────────────────────────────────────
# Start from base cases, fill table in order.
# No recursion overhead, explicit space control.
# Usually faster in practice due to no call stack.

def fib_tab(n):
    if n <= 1:
        return n
    dp = [0] * (n + 1)
    dp[0], dp[1] = 0, 1
    for i in range(2, n + 1):
        dp[i] = dp[i-1] + dp[i-2]
    return dp[n]

# Space-optimized: only need last 2 values
def fib_optimal(n):
    if n <= 1: return n
    prev2, prev1 = 0, 1
    for _ in range(2, n + 1):
        prev2, prev1 = prev1, prev2 + prev1
    return prev1
Memoization (top-down)Tabulation (bottom-up)
DirectionRecursive, start from answerIterative, build from base cases
Code styleCloser to brute-force recursionArray/dict filling loop
SpaceMemo + call stack O(n)Just table O(n), often optimizable
SpeedSlightly slower (function call overhead)Usually faster
When to useWhen state space is sparse or complexWhen all subproblems need to be solved
Core Patterns

1D DP Patterns

Pattern: Linear Recurrence

dp[i] depends on a fixed number of previous states.

Python
# CLIMBING STAIRS: n steps, can climb 1 or 2 at a time
# dp[i] = ways to reach step i = dp[i-1] + dp[i-2]
def climbStairs(n: int) -> int:
    if n <= 2: return n
    a, b = 1, 2
    for _ in range(3, n + 1):
        a, b = b, a + b
    return b

# HOUSE ROBBER: rob non-adjacent houses for max money
# dp[i] = max money robbing from first i houses
# dp[i] = max(dp[i-1], dp[i-2] + nums[i])
def rob(nums: list) -> int:
    if not nums: return 0
    if len(nums) == 1: return nums[0]
    prev2, prev1 = 0, 0
    for num in nums:
        prev2, prev1 = prev1, max(prev1, prev2 + num)
    return prev1

Pattern: Unbounded Choices (Coin Change)

Python
# COIN CHANGE: min coins to make amount (unlimited supply of each coin)
# dp[i] = min coins to make amount i
# dp[i] = min(dp[i], dp[i - coin] + 1) for each coin
def coinChange(coins: list, amount: int) -> int:
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0  # base case: 0 coins to make amount 0

    for i in range(1, amount + 1):
        for coin in coins:
            if coin <= i:
                dp[i] = min(dp[i], dp[i - coin] + 1)

    return dp[amount] if dp[amount] != float('inf') else -1

# COIN CHANGE II: count ways to make amount
# dp[i] += dp[i - coin]
def change(amount: int, coins: list) -> int:
    dp = [0] * (amount + 1)
    dp[0] = 1  # one way to make 0: use no coins
    for coin in coins:
        for i in range(coin, amount + 1):
            dp[i] += dp[i - coin]
    return dp[amount]

Pattern: Subsequence DP

Python
# LONGEST INCREASING SUBSEQUENCE (LIS)
# dp[i] = length of LIS ending at index i
# For each j < i where nums[j] < nums[i]: dp[i] = max(dp[i], dp[j]+1)
def lengthOfLIS(nums: list) -> int:
    n = len(nums)
    dp = [1] * n  # each element alone is LIS of length 1
    for i in range(1, n):
        for j in range(i):
            if nums[j] < nums[i]:
                dp[i] = max(dp[i], dp[j] + 1)
    return max(dp)

# Optimal LIS with patience sort: O(n log n)
import bisect
def lengthOfLIS_fast(nums: list) -> int:
    tails = []  # tails[i] = smallest tail of LIS of length i+1
    for num in nums:
        pos = bisect.bisect_left(tails, num)
        if pos == len(tails):
            tails.append(num)
        else:
            tails[pos] = num
    return len(tails)
2D Problems

2D DP Patterns

Pattern: String Comparison (LCS / Edit Distance)

Python
# LONGEST COMMON SUBSEQUENCE
# dp[i][j] = LCS length for s1[:i] and s2[:j]
def longestCommonSubsequence(s1: str, s2: str) -> int:
    m, n = len(s1), len(s2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if s1[i-1] == s2[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1   # characters match: extend LCS
            else:
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])  # skip one

    return dp[m][n]

# EDIT DISTANCE (Levenshtein)
# dp[i][j] = min operations to convert word1[:i] to word2[:j]
def minDistance(word1: str, word2: str) -> int:
    m, n = len(word1), len(word2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    for i in range(m + 1): dp[i][0] = i   # delete all of word1
    for j in range(n + 1): dp[0][j] = j   # insert all of word2

    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if word1[i-1] == word2[j-1]:
                dp[i][j] = dp[i-1][j-1]       # no operation needed
            else:
                dp[i][j] = 1 + min(
                    dp[i-1][j],    # delete from word1
                    dp[i][j-1],    # insert into word1
                    dp[i-1][j-1]   # replace
                )
    return dp[m][n]

Pattern: Grid DP

Python
# UNIQUE PATHS: count paths from top-left to bottom-right (only right/down)
# dp[i][j] = paths to reach (i,j) = dp[i-1][j] + dp[i][j-1]
def uniquePaths(m: int, n: int) -> int:
    dp = [[1] * n for _ in range(m)]
    for i in range(1, m):
        for j in range(1, n):
            dp[i][j] = dp[i-1][j] + dp[i][j-1]
    return dp[m-1][n-1]

# MINIMUM PATH SUM in grid
def minPathSum(grid: list) -> int:
    m, n = len(grid), len(grid[0])
    dp = [[0] * n for _ in range(m)]
    dp[0][0] = grid[0][0]
    for i in range(1, m): dp[i][0] = dp[i-1][0] + grid[i][0]
    for j in range(1, n): dp[0][j] = dp[0][j-1] + grid[0][j]
    for i in range(1, m):
        for j in range(1, n):
            dp[i][j] = grid[i][j] + min(dp[i-1][j], dp[i][j-1])
    return dp[m-1][n-1]

Pattern: 0/1 Knapsack

Python
# 0/1 KNAPSACK: pick items (each used at most once) to maximize value within capacity
# dp[i][w] = max value using first i items with capacity w
def knapsack(weights, values, capacity):
    n = len(weights)
    dp = [[0] * (capacity + 1) for _ in range(n + 1)]
    for i in range(1, n + 1):
        for w in range(capacity + 1):
            # Don't take item i-1
            dp[i][w] = dp[i-1][w]
            # Take item i-1 if it fits
            if weights[i-1] <= w:
                dp[i][w] = max(dp[i][w], dp[i-1][w - weights[i-1]] + values[i-1])
    return dp[n][capacity]

# Space-optimized: O(capacity) instead of O(n*capacity)
# Must iterate backwards to prevent reusing items
def knapsack_optimized(weights, values, capacity):
    dp = [0] * (capacity + 1)
    for i in range(len(weights)):
        for w in range(capacity, weights[i] - 1, -1):  # backwards!
            dp[w] = max(dp[w], dp[w - weights[i]] + values[i])
    return dp[capacity]
Key Skill

How to Design DP State

The hardest part of DP is identifying what information needs to be in the state. Follow this process:

  1. Write brute-force recursion first — what parameters change between calls?
  2. Those parameters are your state — dp[param1][param2]
  3. Define dp[state] clearly — "dp[i] = max profit up to day i" NOT "dp[i] = something"
  4. Write the recurrence — how does dp[i] relate to dp[i-1], dp[i-2], etc.?
  5. Identify base cases — what are the trivially-known answers?
  6. Determine iteration order — must fill states before they're needed

Linear (1D)

Fibonacci, Climbing Stairs, House Robber, Coin Change. dp[i] depends on dp[i-1], dp[i-2], or earlier.

Subsequence

LIS, LCS. dp[i] or dp[i][j] captures best subsequence ending at position i (or i,j for two strings).

Grid/Matrix

Unique Paths, Min Path Sum. dp[r][c] = answer for reaching cell (r,c). Fill left-to-right, top-to-bottom.

Interval

Burst Balloons, Matrix Chain. dp[i][j] = answer for subproblem on interval [i,j]. Need all shorter intervals first.

State Machine

Best Time to Buy/Sell Stock. Multiple states (holding, not holding, cooldown). Each state has its own DP array.

Bitmask DP

Traveling Salesman. dp[mask][v] = min cost to visit nodes in mask, ending at v. For n ≤ 20.

Optimization

Space Optimization

Many 2D DP tables can be compressed to 1D (or rolling array) because each row only depends on the previous row.

Python
# LCS: reduce O(m*n) to O(n) space
def lcs_optimized(s1: str, s2: str) -> int:
    m, n = len(s1), len(s2)
    prev = [0] * (n + 1)

    for i in range(1, m + 1):
        curr = [0] * (n + 1)
        for j in range(1, n + 1):
            if s1[i-1] == s2[j-1]:
                curr[j] = prev[j-1] + 1
            else:
                curr[j] = max(prev[j], curr[j-1])
        prev = curr

    return prev[n]

# General rule: if dp[i][j] only needs dp[i-1][j] and dp[i-1][j-1] and dp[i][j-1],
# you can use a single 1D array, but must be careful about iteration order.

LC #322Coin ChangeMedium

Minimum coins to make amount. Classic unbounded knapsack / BFS on implicit graph.

Python
def coinChange(self, coins: list[int], amount: int) -> int:
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0
    for i in range(1, amount + 1):
        for coin in coins:
            if coin <= i:
                dp[i] = min(dp[i], dp[i - coin] + 1)
    return dp[amount] if dp[amount] != float('inf') else -1
⏱ Time: O(amount · |coins|)💾 Space: O(amount)

LC #139Word BreakMedium

Can string s be segmented into words from wordDict? dp[i] = can s[:i] be segmented.

Python
def wordBreak(self, s: str, wordDict: list[str]) -> bool:
    words = set(wordDict)
    n = len(s)
    dp = [False] * (n + 1)
    dp[0] = True  # empty string can always be segmented

    for i in range(1, n + 1):
        for j in range(i):
            if dp[j] and s[j:i] in words:
                dp[i] = True
                break
    return dp[n]
⏱ Time: O(n²)💾 Space: O(n)
LC #416Partition Equal Subset SumMedium

Can array be partitioned into two equal-sum subsets? = 0/1 knapsack: can we pick elements summing to total/2?

Python
def canPartition(self, nums: list[int]) -> bool:
    total = sum(nums)
    if total % 2 != 0:
        return False
    target = total // 2
    dp = {0}  # set of achievable sums
    for num in nums:
        dp = {s + num for s in dp} | dp
        if target in dp:
            return True
    return target in dp
⏱ Time: O(n · target)💾 Space: O(target)
